#### 1.1用户信息 (User Information) 索引
创建索引
```
PUT /user_information
{
  "mappings": {
    "properties": {
      "user_id": {
        "type": "keyword"
      },
      "name": {
        "type": "text",
        "analyzer": "standard"
      },
      "email": {
        "type": "keyword"
      },
      "date_of_birth": {
        "type": "date"
      },
      "gender": {
        "type": "keyword"
      },
      "address": {
        "type": "text",
        "analyzer": "standard"
      },
      "phone_number": {
        "type": "keyword"
      },
      "registration_date": {
        "type": "date"
      },
      "last_login": {
        "type": "date"
      },
      "status": {
        "type": "keyword"
      }
    }
  }
}
```

```

## 四、问题与解决
问题：items字段没有定义为一个嵌套字段，而是被定义为一个常规对象。不能使用嵌套查询来查找包含特定product_id的订单。
解决：1. 删除旧索引；2. 创建新的索引映射；3. 重新导入数据；4. 执行查询。
代码：
```
#删除旧索引
DELETE /orders
#创建新的索引映射
PUT /orders
{
  "mappings": {
    "properties": {
      "customer_id": {
        "type": "text",
        "fields": {
          "keyword": {
            "type": "keyword",
            "ignore_above": 256
          }
        }
      },
      "delivery_date": {
        "type": "date"
      },
      "items": {
        "type": "nested",
        "properties": {
          "price": {
            "type": "float"
          },
          "product_id": {
            "type": "text",
            "fields": {
              "keyword": {
                "type": "keyword",
                "ignore_above": 256
              }
            }
          },
          "quantity": {
            "type": "long"
          }
        }
      },
      "order_date": {
        "type": "date"
      },
      "order_id": {
        "type": "text",
        "fields": {
          "keyword": {
            "type": "keyword",
            "ignore_above": 256
          }
        }
      },
      "payment_method": {
        "type": "text",
        "fields": {
          "keyword": {
            "type": "keyword",
            "ignore_above": 256
          }
        }
      },
      "shipping_address": {
        "type": "text",
        "fields": {
          "keyword": {
            "type": "keyword",
            "ignore_above": 256
          }
        }
      },
      "shipping_date": {
        "type": "date"
      },
      "status": {
        "type": "text",
        "fields": {
          "keyword": {
            "type": "keyword",
            "ignore_above": 256
          }
        }
      },
      "total_amount": {
        "type": "float"
      }
    }
  }
}
#执行查询
GET /orders/_search
{
  "query": {
    "nested": {
      "path": "items",

        "bool": {
          "must": {
            "term": {
              "items.product_id": "P001"
            }
          }
        }
      }
    }
  }
}
